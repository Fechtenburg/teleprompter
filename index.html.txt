let words        = [];   // normalized words for matching
let wordEls      = [];   // DOM spans

let isListening  = false;
let isMirrored   = false;

let recognition  = null;
let restartTimer = null;
let fadeTimer    = null;

const LANGS = [
  { code: 'da-DK', label: 'DA' },
  { code: 'en-US', label: 'EN' },
  { code: 'no-NO', label: 'NO' },
  { code: 'sv-SE', label: 'SV' },
  { code: 'de-DE', label: 'DE' },
];
let langIdx = 0;

/* ════════════════════════════════
   BUILD WORD SPANS
════════════════════════════════ */
function normalize(w) {
  return w.toLowerCase().replace(/[^a-zæøåéàüöä0-9]/g, '');
}

function buildSpans(text) {
  const container = document.getElementById('prompter-text');
  container.innerHTML = '';
  wordEls = [];
  words   = [];

  text.split(/(\n+)/).forEach(part => {
    if (/^\n+$/.test(part)) {
      container.appendChild(document.createElement('br'));
      if (part.length > 1) container.appendChild(document.createElement('br'));
    } else {
      part.split(/(\s+)/).forEach(tok => {
        if (/^\s+$/.test(tok)) {
          container.appendChild(document.createTextNode(' '));
        } else if (tok) {
          const span = document.createElement('span');
          span.className = 'word future';
          span.textContent = tok;
          container.appendChild(span);
          words.push(normalize(tok));
          wordEls.push(span);
        }
      });
    }
  });
}

/* ════════════════════════════════
   SCROLL & HIGHLIGHT
════════════════════════════════ */
function applyHighlight(idx) {
  wordEls.forEach((el, i) => {
    el.className = 'word ' + (i < idx ? 'past' : i === idx ? 'active' : 'future');
  });
  const pct = words.length > 1 ? Math.round((idx / (words.length - 1)) * 100) : 0;
  document.getElementById('progress-bar').style.width = pct + '%';
  document.getElementById('hud-pct').textContent = pct + '%';
}

function scrollToIdx(idx) {
  const el = wordEls[idx];
  if (!el) return;
  const screenH = document.getElementById('prompter-screen').offsetHeight;
  const targetY = el.offsetTop - screenH * 0.40;
  document.getElementById('scroll-container').style.transform =
    `translateY(${-Math.max(0, targetY)}px)`;
}

function setIdx(idx) {
  idx = Math.max(0, Math.min(words.length - 1, idx));
  currentIdx = idx;
  applyHighlight(idx);
  scrollToIdx(idx);
}

function jumpWords(n) {
  setIdx(currentIdx + n);
  showControls();
}

/* ════════════════════════════════
   FUZZY WORD MATCHING
   Dice bigram similarity
════════════════════════════════ */
function diceSim(a, b) {
  if (!a || !b) return 0;
  if (a === b) return 1;
  if (a.length < 2 || b.length < 2) return a[0] === b[0] ? 0.65 : 0;
  const bigrams = s => {
    const m = {};
    for (let i = 0; i < s.length - 1; i++) {
      const k = s[i] + s[i+1];
      m[k] = (m[k] || 0) + 1;
    }
    return m;
  };
  const A = bigrams(a), B = bigrams(b);
  let hits = 0;
  Object.keys(A).forEach(k => { if (B[k]) hits += Math.min(A[k], B[k]); });
  return (2 * hits) / (a.length + b.length - 2);
}

/* ════════════════════════════════
   SPEECH → WORD POSITION
   Match each final transcript against
   the script, advancing currentIdx
════════════════════════════════ */
function matchTranscript(transcript, isFinal) {
  const heard = transcript.toLowerCase()
    .replace(/[^a-zæøåéàüöä0-9\s]/g, '')
    .trim().split(/\s+/).filter(w => w.length > 0);

  if (!heard.length) return;

  // For interim: only match the last heard word (fast feedback)
  // For final: walk through all heard words sequentially
  if (isFinal) {
    let searchFrom = currentIdx;
    heard.forEach(hw => {
      const WINDOW = 25;
      let bestIdx = -1, bestScore = 0;
      const end = Math.min(words.length, searchFrom + WINDOW);
      for (let i = searchFrom; i < end; i++) {
        const s = diceSim(hw, words[i]);
        if (s > bestScore && s > 0.58) { bestScore = s; bestIdx = i; }
      }
      if (bestIdx !== -1) {
        setIdx(bestIdx);
        searchFrom = bestIdx + 1;
      }
    });
  } else {
    // Interim: match only last heard word for snappy response
    const hw = heard[heard.length - 1];
    const WINDOW = 20;
    let bestIdx = -1, bestScore = 0;
    const end = Math.min(words.length, currentIdx + WINDOW);
    for (let i = currentIdx; i < end; i++) {
      const s = diceSim(hw, words[i]);
      if (s > bestScore && s > 0.65) { bestScore = s; bestIdx = i; }
    }
    if (bestIdx !== -1 && bestIdx > currentIdx) {
      setIdx(bestIdx);
    }
  }
}

/* ════════════════════════════════
   SPEECH RECOGNITION
════════════════════════════════ */
function buildRecognition() {
  const SR = window.SpeechRecognition || window.webkitSpeechRecognition;
  if (!SR) return null;

  const r = new SR();
  r.continuous      = true;
  r.interimResults  = true;
  r.maxAlternatives = 1;
  r.lang            = LANGS[langIdx].code;

  r.onstart = () => setHudState('listening');

  r.onresult = e => {
    let interim = '';
    for (let i = e.resultIndex; i < e.results.length; i++) {
      const t = e.results[i][0].transcript;
      if (e.results[i].isFinal) {
        matchTranscript(t, true);
        document.getElementById('interim-display').textContent = '';
      } else {
        interim += t;
        matchTranscript(t, false);
      }
    }
    if (interim) document.getElementById('interim-display').textContent = interim;
  };

  r.onspeechstart = () => setHudState('listening');
  r.onspeechend   = () => {
    setHudState('paused');
    document.getElementById('interim-display').textContent = '';
  };

  r.onerror = e => {
    if (e.error === 'no-speech' || e.error === 'aborted') return;
    console.warn('SR:', e.error);
    setHudState('paused');
  };

  r.onend = () => {
    if (isListening) {
      restartTimer = setTimeout(() => {
        try { recognition.start(); } catch(ex) {}
      }, 250);
    }
  };

  return r;
}

function setHudState(state) {
  const dot = document.getElementById('mic-dot');
  const lbl = document.getElementById('hud-status');
  dot.className = 'mic-dot';
  if (state === 'listening') {
    dot.classList.add('listening');
    lbl.textContent = 'Lytter…';
  } else if (state === 'paused') {
    dot.classList.add('paused');
    lbl.textContent = 'Pause — venter…';
  } else {
    lbl.textContent = 'Klar';
  }
}

function toggleListening() {
  isListening ? stopListening() : startListening();
  showControls();
  if (isListening) scheduleControlsFade();
}

function startListening() {
  if (!recognition) recognition = buildRecognition();
  if (!recognition) {
    alert('Stemmegenkendelse ikke tilgængeligt.\n\nBrug Safari på iOS 15+ eller Chrome på Android.');
    return;
  }
  isListening = true;
  recognition.lang = LANGS[langIdx].code;
  document.getElementById('mic-btn').classList.add('active');
  document.getElementById('mic-btn').textContent = '⏹';
  try { recognition.start(); } catch(e) {}
}

function stopListening() {
  isListening = false;
  clearTimeout(restartTimer);
  document.getElementById('mic-btn').classList.remove('active');
  document.getElementById('mic-btn').textContent = '🎙';
  document.getElementById('interim-display').textContent = '';
  setHudState('idle');
  try { recognition.stop(); } catch(e) {}
}

/* ════════════════════════════════
   CONTROLS FADE
════════════════════════════════ */
function showControls() {
  document.getElementById('prompter-controls').classList.remove('faded');
  document.getElementById('hud').classList.remove('faded');
  clearTimeout(fadeTimer);
}
function scheduleControlsFade() {
  clearTimeout(fadeTimer);
  fadeTimer = setTimeout(() => {
    if (isListening) {
      document.getElementById('prompter-controls').classList.add('faded');
      document.getElementById('hud').classList.add('faded');
    }
  }, 4000);
}

document.getElementById('prompter-screen').addEventListener('click', e => {
  if (e.target.closest('.ctrl-btn')) return;
  if (document.getElementById('prompter-controls').classList.contains('faded')) {
    showControls();
    if (isListening) scheduleControlsFade();
  }
});

let touchStartY = null, touchStartIdx = 0;
const ps = document.getElementById('prompter-screen');

ps.addEventListener('touchstart', e => {
  if (e.target.closest('.ctrl-btn')) return;
  touchStartY = e.touches[0].clientY;
  touchStartIdx = currentIdx;
}, { passive: true });

ps.addEventListener('touchmove', e => {
  if (touchStartY === null || e.target.closest('.ctrl-btn')) return;
  const dy = touchStartY - e.touches[0].clientY;
  const delta = Math.round(dy / (fontSize * 1.6 * 0.35));
  setIdx(touchStartIdx + delta);
}, { passive: true });

ps.addEventListener('touchend', () => { touchStartY = null; }, { passive: true });

function startPrompter() {
  const text = document.getElementById('script-input').value.trim();
  if (!text) { alert('Indsæt venligst et manuskript først.'); return; }

  buildSpans(text);
  currentIdx = 0;

  document.documentElement.style.setProperty('--font-size', fontSize + 'px');

  isMirrored
    ? document.getElementById('prompter-text').style.transform = 'scaleX(-1)'
    : document.getElementById('prompter-text').style.transform = '';

  document.getElementById('editor-screen').classList.add('hidden');
  document.getElementById('prompter-screen').classList.remove('hidden');
  document.getElementById('scroll-container').style.transform = 'translateY(0)';
  applyHighlight(0);

  showControls();
  setTimeout(() => startListening(), 400);
  scheduleControlsFade();
}

  stopListening();
  recognition = null;
  clearTimeout(fadeTimer);
  document.getElementById('prompter-screen').classList.add('hidden');
  document.getElementById('editor-screen').classList.remove('hidden');
}
